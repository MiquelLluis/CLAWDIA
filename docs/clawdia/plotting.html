clawdia.plotting

Ad-hoc plotting functions for visualizing results.

This module includes a variety of plotting utilities designed to help visualize and interpret results during the development and debugging of the CLAWDIA pipeline.

While not essential for CLAWDIA's core processing, these functions are useful for presenting and analyzing outcomes, such as confusion matrices, dictionary atoms, and spectrograms.

 46def plot_confusion(cmat, ax=None, labels=None, mode='both', vmin=None, vmax=None,
 47                   cmap="PaleBlues", **kwargs):
 48    """Plot a confusion matrix.
 49
 50    Plot a pre-computed confusion matrix `cmat`.
 51    Rows must contain true values, and columns predicted values. For example,
 52    in a binary classification case:
 53
 54    | T \ P  | Pred 0  | Pred 1  |
 55    |--------|---------|---------|
 56    | True 0 | TP      | FN      |
 57    | True 1 | FP      | TN      |
 58
 59    Where:
 60    - **TP** (True Positives): Correctly predicted Class 0.
 61    - **FN** (False Negatives): Class 0 incorrectly predicted as Class 1.
 62    - **FP** (False Positives): Class 1 incorrectly predicted as Class 0.
 63    - **TN** (True Negatives): Correctly predicted Class 1.
 64
 65    Parameters
 66    ----------
 67    cmat : array-like of shape (n_classes, n_classes)
 68        The confusion matrix to plot.
 69    
 70    ax : matplotlib.axes.Axes, optional
 71        The axes on which to plot the matrix. If not given, a new figure and axes
 72        are created.
 73    
 74    labels : list of str, optional
 75        The labels for the classes. If not given, the integers from 0 to
 76        `n_classes-1` are used.
 77    
 78    mode : {'absolute', 'percent', 'both'}
 79        The format of the annotations: absolute numbers, percentages, or both.
 80        Defaults to 'both'.
 81    
 82    vmin : float, optional
 83        The minimum value of the color scale.
 84    
 85    vmax : float, optional
 86        The maximum value of the color scale.
 87
 88    cmap : Function (matplotlib's Colormap equivalent) | str
 89        Defaults to "PaleBlue", a custom modification of Matplotlib's "Blues".
 90    
 91    **kwargs
 92        Additional keyword arguments are passed to `matplotlib.pyplot.subplots`.
 93
 94    Returns
 95    -------
 96    fig : matplotlib.figure.Figure or None
 97        The figure containing the plot, or None if `ax` was given.
 98
 99    """
100    with np.errstate(divide='ignore', invalid='ignore'):
101        cmat_perc = np.nan_to_num(cmat / np.sum(cmat, axis=1, keepdims=True))
102    
103    if mode == 'both':
104        if plt.rcParams["text.usetex"]:
105            format_str = lambda *args: '{}\n{:.0%}'.format(*args).replace('%', r'\,\%')
106        else:
107            format_str = lambda *args: '{}\n{:.0%}'.format(*args)
108    elif mode == 'percent':
109        if plt.rcParams["text.usetex"]:
110            format_str = lambda *args: '{:.0%}'.format(args[1]).replace('%', r'\,\%')
111        else:
112            format_str = lambda *args: '{:.0%}'.format(args[1])
113    elif mode == 'absolute':
114        format_str = lambda *args: str(args[0])
115    else:
116        raise ValueError("mode can only be 'absolute', 'percent' or 'both'")
117    
118    if not callable(cmap):
119        if cmap == "PaleBlues":
120            cmap = _desaturate_cmap(plt.cm.Blues, desaturation_factor=0.8, brightness_boost=0.2)
121        elif isinstance(cmap, str):
122            cmap = plt.get_cmap(cmap)
123        else:
124            raise TypeError("Colormap not recognized")
125
126    n_classes = len(cmat)
127
128    cmat_str = np.empty_like(cmat_perc, dtype=object)
129    for i_true, i_pred in it.product(range(n_classes), repeat=2):
130        cmat_str[i_true, i_pred] = format_str(cmat[i_true, i_pred], cmat_perc[i_true, i_pred])
131
132    if ax is None:
133        fig, ax = plt.subplots(**kwargs)
134
135    ax.imshow(cmat_perc, cmap=cmap, vmin=vmin, vmax=vmax)
136    
137    for i_true, i_pred in it.product(range(n_classes), repeat=2):
138        ax.annotate(cmat_str[i_true, i_pred], xy=(i_pred, i_true), ha='center', va='center')
139    
140    ax.grid(False)
141
142    # Set X and Y labels
143    ax.set_xlabel("Predicted")
144    ax.set_ylabel("True")
145
146    # Set ticks
147    ax.set_xticks(range(n_classes))
148    ax.set_yticks(range(n_classes))
149    
150    ax.set_xlim([-0.5, n_classes-0.5])
151    ax.set_ylim([n_classes-0.5, -0.5])  # Invert Y-axis
152
153    if labels is not None:
154        ax.set_xticklabels(labels)
155        ax.set_yticklabels(labels)
156
157    # Move X-axis ticks and label to the top
158    ax.xaxis.tick_top()
159    ax.xaxis.set_label_position('top')
160
161    try:
162        return fig
163    except NameError:
164        return None

168def plot_dictionary(array, c=None, ylim=None, **plot_kw):
169    """Plot atoms from a dictionary in a squared matrix.
170    
171    Parameters
172    ----------
173    array : 2d-array
174        Dictionary matrix in Fortran order with shape (a, l).
175    
176    c : int, optional
177        Number of atoms at each side of the squared matrix of plots; the total
178        number of plotted atoms will be `c ** 2`.
179        If not given, it is computed as `int(np.sqrt(a))`.
180    
181    **plot_kw : optional
182        Passed to pyplot.subplots().
183    
184    """
185    if c is None:
186        c = int(np.sqrt(array.shape[0]))
187    fig, axs = plt.subplots(ncols=c, nrows=c, **plot_kw)
188    for i in range(c**2):
189        ax = axs[i//c,i%c]
190        ax.plot(array[i], lw=1)
191        ax.set_ylim(ylim)
192        ax.axis('off')
193    fig.tight_layout()
194    fig.subplots_adjust(wspace=0, hspace=0)
195    
196    return fig

199def plot_spec_of(strain, figsize=(10,5), sf=4096, window='hann', vmin=None, vmax=None):
200    fig, ax = plt.subplots(figsize=figsize)
201    ff, tt, spec = sp.signal.spectrogram(
202        strain,
203        fs=sf,
204        window=window,
205        nperseg=256,
206        nfft=2*sf,
207        noverlap=256-32
208    )
209    norm = mpl.colors.LogNorm(clip=False, vmin=vmin, vmax=vmax)
210    pcm = ax.pcolormesh(tt, ff, spec, norm=norm)
211
212    return fig, spec, pcm

215def plot_spectrogram_with_instantaneous_features(
216        strain_array, time_array, sampling_rate=2**14, outseg=None, outfreq=None,
217        window=sp.signal.windows.tukey(128,0.5), hop=32, mfft=2**14, vmin=-22, ax=None):
218    """
219    Plot the spectrogram, instantaneous frequency, and strain's waveform.
220
221    This function generates a multi-panel plot consisting of:
222    
223    1. A spectrogram of the gravitational wave strain, obtained using
224    Short-Time Fourier Transform (STFT), visualizing the frequency evolution
225    over time.
226    2. The instantaneous frequency of the strain, plotted on top of the
227    spectrogram to show the frequency changes in real time.
228    3. The raw gravitational wave strain in the time domain, shown above the
229    spectrogram for direct comparison.
230
231    Key features of the plot:
232    - **Spectrogram**: The frequency content of the gravitational wave signal
233      is displayed over time using a color map (`inferno`), with the x-axis
234      representing time (in milliseconds) and the y-axis representing
235      frequency (in Hz).
236    - **Instantaneous Frequency**: Plots the instantaneous frequency of the
237      strain over time, highlighting the frequency variations.
238    - **Energy Normalization**: The spectrogram uses a logarithmic scale for
239      the energy (power spectral density, PSD), normalized by the maximum
240      energy value in the signal.
241    - **Dynamic Range Control**: The color scale of the spectrogram can be
242      adjusted via the `vmin` parameter to emphasize specific energy levels.
243    - **Time-Domain Waveform**: A plot of the original strain data in the time
244      domain is shown above the spectrogram, providing context for the signal's
245      evolution.
246    - **Segmentation**: The user can specify the time (`outseg`) and frequency
247      (`outfreq`) ranges to focus on specific parts of the data.
248    - **Customization**: The plot has a black background, white grid lines,
249      and labeled colorbars for clarity.
250
251    Parameters
252    ----------
253    strain_array : numpy.ndarray
254        The time-domain strain data of the gravitational wave signal.
255    
256    time_array : numpy.ndarray
257        Array of time stamps corresponding to the strain data.
258    
259    sampling_rate : int, optional
260        The sampling rate of the data in Hz (default is 2^14, or 16384 Hz).
261    
262    outseg : tuple, optional
263        A tuple specifying the time range (start, end) in seconds for the
264        x-axis. If `None`, the entire time range of the input data is used.
265    
266    outfreq : tuple, optional
267        A tuple specifying the frequency range (start, end) in Hz for the
268        y-axis. If `None`, the full frequency range (up to Nyquist frequency)
269        is used.
270    
271    window : numpy.ndarray, optional
272        The window function applied during STFT computation (default is a
273        Tukey window).
274    
275    hop : int, optional
276        The hop size between successive STFT windows (default is 32).
277    
278    mfft : int, optional
279        The number of points in the FFT used for STFT computation (default
280        is 2^14).
281    
282    vmin : float, optional
283        The minimum value for the color scale in the spectrogram (default
284        is -22). This controls the dynamic range of the color map.
285    
286    ax : matplotlib.axes.Axes, optional
287        The axes object on which to plot the spectrogram. If `None`, a new
288        figure and axes are created.
289    
290    Returns
291    -------
292    fig : matplotlib.figure.Figure
293        The figure object containing the complete plot.
294    
295    axs : List[matplotlib.axes.Axes]
296        A list of axes objects containing the spectrogram, the colorbar, and
297        the time-domain plots.
298    
299    Sxx : numpy.ndarray
300        The computed spectrogram (PSD values) of the input strain data.
301
302    Notes
303    -----
304    - The spectrogram normalization is performed on the square root of the 
305      Power Spectral Density (PSD), converted to a logarithmic scale.
306    - The y-axis of the spectrogram uses a kilohertz scale for readability.
307    - The time-domain waveform is plotted without axes labels for simplicity.
308    - Instantaneous frequency values below zero are masked to avoid displaying 
309      non-physical results.
310    
311    """
312    from gwadama.fat import instant_frequency
313
314    # Compute the spectrogram using the ShortTimeFFT class.
315    stfft_model = sp.signal.ShortTimeFFT(
316        win=window, hop=hop, fs=sampling_rate, mfft=mfft,
317        fft_mode='onesided', scale_to='psd'
318    )
319    Sxx = stfft_model.spectrogram(strain_array)
320    with np.errstate(divide='ignore'):
321        normalized_Sxx = np.log10(np.sqrt(Sxx))
322    normalized_Sxx -= np.max(normalized_Sxx)
323    t0, t1, f0, f1 = stfft_model.extent(len(strain_array))
324    t_origin = time_array[0]
325    t0 += t_origin
326    t1 += t_origin
327
328    if ax is None:
329        fig, ax = plt.subplots()
330    else:
331        fig = ax.get_figure()
332
333    # SPECTROGRAM (ax1)
334    ax.imshow(normalized_Sxx, cmap='inferno', origin='lower', aspect='auto',
335              extent=(t0,t1,f0,f1), interpolation='lanczos', vmin=vmin)
336    # ...and Instant Frequency
337    with warnings.catch_warnings():
338        warnings.simplefilter("ignore", RuntimeWarning)
339        instant_freq = instant_frequency(strain_array, sample_rate=sampling_rate)
340    length = len(strain_array)
341    t1_instant = t_origin + (length-1)/sampling_rate
342    instant_time = np.linspace(t_origin, t1_instant, length)
343    mask = instant_freq >= 0  # Remove non-physical frequencies
344    instant_freq = instant_freq[mask]
345    instant_time = instant_time[mask]
346    ax.plot(instant_time, instant_freq, 'purple', lw=2)
347    
348    # COLORBAR (ax2)
349    ax_pos = ax.get_position()
350    ax2_width = 0.015  # Set the width of the colorbar axis
351    ax2_pad = 0.01    # Set the padding between the spectrogram and colorbar
352    ax2_x = ax_pos.x1 + ax2_pad
353    ax2 = fig.add_axes([ax2_x, ax_pos.y0, ax2_width, ax_pos.height])
354    cbar = fig.colorbar(ax.images[-1], cax=ax2)
355    
356    # LABELS, LIMITS, ETC
357    ax.grid(True, ls='--', alpha=.4)
358    # ...limits
359    if outseg is None:
360        ax.set_xlim(time_array[0], time_array[-1])
361    else:
362        ax.set_xlim(*outseg)
363    if outfreq is None:
364        ax.set_ylim(0, sampling_rate/2)
365    else:
366        ax.set_ylim(*outfreq)
367    # ...labels.
368    ax.set_xlabel('Time [ms]')
369    ax.set_ylabel('Frequency [kHz]')
370    cbar.set_label('Normalized energy')
371    # ...Y ticks to kHz
372    ax.yaxis.set_major_formatter(mpl.ticker.FuncFormatter(lambda x, pos: f'{x / 1000:.0f}'))
373    # ...X ticks to milliseconds and avoid roundoff errors.
374    # ax.xaxis.set_major_formatter(FormatStrFormatter('%d'))  # Set style first, for new ticklabels.
375    xticks = ax.get_xticks()
376    ax.set_xticks(xticks)
377    # ax.set_xticklabels(np.round(xticks * 1000).astype(int))
378    ax.set_xticklabels(np.round(xticks*1000, decimals=1))
379    # ...background in black to match the colormap.
380    ax.set_facecolor('black')
381
382    # GW IN TIME-DOMAIN ON TOP OF THE SPECTROGRAM (ax3)
383    ax3 = fig.add_axes([ax_pos.x0, ax_pos.y0+ax_pos.height+0.03, ax_pos.width, ax_pos.height*0.2])
384    ax3.plot(time_array, strain_array, c='black', lw=1, alpha=1)
385    ax3.set_xlim(ax.get_xlim())
386    ax3.set_ylim(np.min(strain_array), np.max(strain_array))
387    ax3.axis('off')
388
389    # # DEBUG AXES AREA [
390    # from matplotlib import patches
391    # ax3_pos = ax3.get_position()
392    # ax2_pos = ax2.get_position()
393    # # Create a rectangle for the main axis (ax)
394    # ax_rect = patches.Rectangle((ax_pos.x0, ax_pos.y0), ax_pos.width, ax_pos.height,
395    #                             linewidth=3, edgecolor='r', facecolor='none', label='ax')
396    # # Create a rectangle for the colorbar axis (ax2)
397    # ax2_rect = patches.Rectangle((ax2_pos.x0, ax2_pos.y0), ax2_pos.width, ax2_pos.height,
398    #                             linewidth=3, edgecolor='b', facecolor='none', label='ax2')
399    # # Create a rectangle for the time-domain axis (ax3)
400    # ax3_rect = patches.Rectangle((ax3_pos.x0, ax3_pos.y0), ax3_pos.width, ax3_pos.height,
401    #                             linewidth=3, edgecolor='g', facecolor='none', label='ax3')
402    # # Add the rectangles to the figure
403    # fig.add_artist(ax_rect)
404    # fig.add_artist(ax2_rect)
405    # fig.add_artist(ax3_rect)
406    # # ]
407    
408    return fig, [ax, ax2, ax3], Sxx

411def _desaturate_cmap(cmap, *, desaturation_factor, brightness_boost):
412    """
413    Desaturate and brighten a colormap.
414
415    Parameters
416    ----------
417    cmap : matplotlib.colors.Colormap
418        The original colormap to modify.
419    desaturation_factor : float
420        The factor by which to reduce the saturation (0 = grayscale, 1 = no change).
421    brightness_boost : float
422        The amount to increase the brightness (lightness) (0 = no change).
423
424    Returns
425    -------
426    LinearSegmentedColormap
427        The modified colormap with reduced saturation and increased brightness.
428    """
429    colors = cmap(np.linspace(0, 1, 256))  # Extract RGB colors
430    modified_colors = []
431
432    for r, g, b, a in colors:  # Process each RGBA color
433        h, l, s = rgb_to_hls(r, g, b)  # Convert to HLS
434        s *= desaturation_factor  # Reduce saturation
435        l = min(l + brightness_boost * (1 - l), 1.0)  # Adjust brightness, ensure no overflow
436        r, g, b = hls_to_rgb(h, l, s)  # Convert back to RGB
437        modified_colors.append((r, g, b, a))  # Add the alpha channel back
438
439    return mpl.colors.LinearSegmentedColormap.from_list("DesaturatedBrightenedBlues", modified_colors)
